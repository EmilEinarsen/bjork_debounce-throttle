/**
 * Function of: 
 * 	Restricting methods, built up on the naitive SetTimeout function.
 * 	And a couple of local variables
 * @type {Function}
 */
function Restrain() {

	let debounceTimer
	this.debounce = (func, delay, cancel) => {
		return new Promise((resolve, reject) => {
			
			clearTimeout(debounceTimer)
			if(cancel) reject('function canceled')
			
			debounceTimer = setTimeout(() => {
				func()
				resolve('function fired')
			}, delay )

		})
	}

	/**
	 * @param {}
	 */
	let throttleTimer
	let firstCall = true
	this.throttle = (func, delay, cancel) => {
		return new Promise(resolve => {

			if(cancel) throttleTimer = undefined
			if(throttleTimer) return
			if(firstCall) {
				firstCall = false
				func()
				resolve('function fired')
			}
			this.debounce(() => firstCall = true, delay)

			throttleTimer = setTimeout(() => {
				func()
				throttleTimer = undefined
				resolve('function fired')
			}, delay)
		})
	}
}
